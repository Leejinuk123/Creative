> ## 1. 프로세싱 코드
  
  ```processing
  import processing.net.*;
Server s;
Clinet c;
  void setup(){
    size(500,500);
     s=new Server(this,12345);
  }
  void draw(){
    c=s.available();
     if(c!=null){
       String m = c.readString();
       m = m.substring(m.length()-1);
       if(m.indexOf('1')==1){fill(255,0,0);}
       if(m.indexOf('0')==0){fill(0,0,255);}
       ellipse(255,255,100,100);
       println(m);
       
     }
  }
  ```  
